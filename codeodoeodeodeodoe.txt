def face_normal(face):
    polyface = cmds.polyListComponentConversion(face, tvf = True)
    xes = cmds.polyNormalPerVertex(polyface, q=True, x =True)
    yes = cmds.polyNormalPerVertex(polyface, q=True, y =True)
    zes = cmds.polyNormalPerVertex(polyface, q=True, z =True)
    divisor = 1.0 / len(xes)
    return sum(xes)* divisor, sum(yes)  * divisor, sum(zes) * divisor
