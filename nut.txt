import maya.cmds as mc

origObjs = mc.ls(sl=True)

for obj in origObjs:
    # get the face normals of the first object
    origFace = mc.polyInfo(obj, fn=True )
    
    # duplicate the object and use conform on the duplicate
    dupObj = mc.duplicate(obj, rr=True)
    dupShape = mc.listRelatives(dupObj, shapes=True)
    mc.polyNormal(normalMode=2, userNormalMode=0, ch=1)
    
    mc.select(clear=True)
    
    # get the face normals of the duplicate object
    dupFace = mc.polyInfo(dupShape, fn=True)
